Standard DOM API
const div = document.createElement('div');
div.className = 'card';
div.textContent = 'Hello World';
div.setAttribute('id', 'intro');
div.onclick = () => console.log('Hi');
document.body.appendChild(div);
JJ (Just JavaScript)
JJHE.fromTag('div')
    .addClass('card')
    .setText('Hello World')
    .setId('intro')
    .onClick(() => console.log('Hi'))
    .appendToBody()
